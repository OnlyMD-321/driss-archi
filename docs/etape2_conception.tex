\subsection{Description des Cas d'Utilisation Principaux}

\paragraph{UC01 — Créer un appel d'offre}
\begin{itemize}
    \item \textbf{Acteur principal :} Responsable des ressources
    \item \textbf{Précondition :} L'utilisateur est authentifié avec le rôle \texttt{responsable}
    \item \textbf{Scénario nominal :}
    \begin{enumerate}
        \item Le responsable accède au formulaire de création
        \item Il saisit le titre, la description, les dates de début et de fin
        \item Il ajoute les besoins en ressources par département
        \item Le système crée l'appel d'offre au statut "ouvert"
        \item Les fournisseurs non blacklistés peuvent consulter l'appel
    \end{enumerate}
    \item \textbf{Postcondition :} Appel d'offre créé, visible par les fournisseurs
\end{itemize}

\paragraph{UC02 — Soumettre une offre}
\begin{itemize}
    \item \textbf{Acteur principal :} Fournisseur
    \item \textbf{Précondition :} Fournisseur inscrit, non blacklisté, appel ouvert
    \item \textbf{Scénario nominal :}
    \begin{enumerate}
        \item Le fournisseur consulte les appels d'offre ouverts
        \item Il sélectionne un appel et remplit le formulaire d'offre
        \item Il indique la date de livraison, la durée de garantie, le prix total
        \item Il détaille le prix unitaire par type de ressource
        \item Le système enregistre l'offre au statut "en attente"
    \end{enumerate}
    \item \textbf{Exception :} Si le fournisseur est blacklisté → erreur 403
\end{itemize}

\paragraph{UC03 — Sélectionner une offre}
\begin{itemize}
    \item \textbf{Acteur principal :} Responsable des ressources
    \item \textbf{Précondition :} Des offres ont été soumises
    \item \textbf{Scénario nominal :}
    \begin{enumerate}
        \item Le responsable visualise toutes les offres triées par prix croissant
        \item Il élimine les fournisseurs non conformes (blacklist)
        \item Il sélectionne l'offre la moins disante
        \item Le système envoie une notification d'acceptation au fournisseur retenu
        \item Le système envoie les notifications de rejet aux autres fournisseurs
        \item L'appel d'offre passe au statut "attribué"
    \end{enumerate}
\end{itemize}

\paragraph{UC04 — Signaler une panne}
\begin{itemize}
    \item \textbf{Acteur principal :} Enseignant ou Chef de département
    \item \textbf{Précondition :} Ressource affectée au département
    \item \textbf{Scénario nominal :}
    \begin{enumerate}
        \item L'enseignant sélectionne la ressource défectueuse
        \item Il saisit une description de la panne
        \item Le système crée un ticket de panne au statut "ouvert"
        \item La ressource passe au statut "maintenance"
        \item Un technicien est notifié pour intervention
    \end{enumerate}
\end{itemize}

\section{Diagramme de Packages}

L'application est organisée en packages suivants :

\subsection{Backend}

\begin{verbatim}
grm-backend/
├── config/          -- Configuration (DB, variables d'env)
├── middleware/      -- Authentification JWT, autorisation RBAC
├── controllers/     -- Logique métier
│   ├── authController
│   ├── tenderController
│   ├── offerController
│   ├── resourceController
│   ├── supplierController
│   ├── breakdownController
│   └── departmentController
├── routes/          -- Définition des routes REST
└── database/        -- Schéma SQL et migrations
\end{verbatim}

\subsection{Frontend}

\begin{verbatim}
grm-frontend/
├── context/         -- État global (AuthContext)
├── services/        -- Client HTTP (Axios)
├── components/      -- Composants réutilisables
│   └── Layout/      -- Navigation, en-tête
└── pages/           -- Pages par domaine fonctionnel
    ├── Tenders/     -- Appels d'offre
    ├── Resources/   -- Ressources
    ├── Suppliers/   -- Fournisseurs
    ├── Breakdowns/  -- Pannes
    └── Departments/ -- Départements
\end{verbatim}
